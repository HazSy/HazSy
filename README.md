    public static void main(String[] args) {
        // Aquí ponemos la nota que queremos revisar

        // Si la nota es 5 o más, está aprobado
        if (nota >= 5) {
            System.out.println("Has aprobado.");
        } else {
            System.out.println("Lo siento has suspendido");
        }
    }
}
